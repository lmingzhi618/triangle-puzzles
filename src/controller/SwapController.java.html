package controller;

import java.awt.Color;
import java.util.Collections;
import java.util.List;

import model.Edge;
import model.Model;
import view.TrianglePuzzleApp;

public class SwapController {
	Model model;
	TrianglePuzzleApp app;
	
	public SwapController(TrianglePuzzleApp app, Model m) {
		this.model = m;
		this.app = app;
	}
	
	public void process() {
		List<Edge> edges = model.isSelectedNodesAvailable();
		
		System.out.printf("size of selected nodes: %d\n", edges.size());
		if (edges.size() < 2) {
			return;
		}
		Edge e0 = edges.get(0);
		Edge e1 = edges.get(1);
		
		
		if (2 == edges.size()) {
			model.swapColor(edges.get(0), edges.get(1));
			System.out.println("select edges: " + e0.getIdx() + ", " + e1.getIdx());
		} else if (3 == edges.size()) {
			Edge e2 = edges.get(2);
			//if (model.isEdgesInSameTriangle(e0, e1, e2)) {
				Collections.sort(edges);
				Color c0 = e0.getColor();
				Color c1 = e1.getColor();
				e0.setColor(e2.getColor());
				e1.setColor(c0);
				e2.setColor(c1);
				System.out.printf("e2: %d", e2.getIdx());
				System.out.println("select edges: " + e0.getIdx() + ", " + e1.getIdx() + ", " + e2.getIdx());
			//} else {
			//	model.swapColor(edges.get(0), edges.get(1));
			//	System.out.println("select edges: " + e0.getIdx() + ", " + e1.getIdx());
			//}
		}
		
		model.clearSelectedNodes();
		model.unselectAllNodes();
		model.updateMove();
		model.updateScore();

		if (model.hasWon()) {
			app.getCongratsLabel().setText("Congratuations!");
			app.getSwapBtn().setEnabled(false);
			app.getUnselectAllBtn().setEnabled(false);
		}
		int nMove = model.getNumberMove();
		int nScore = model.getScore();
		app.getMoveField().setText(String.valueOf(nMove));;
		app.getScoreField().setText(String.valueOf(nScore));;
		app.getPanel().repaint();
	}
}
