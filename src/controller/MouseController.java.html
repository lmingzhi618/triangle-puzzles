package controller;

import java.awt.Point;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;

import model.Model;
import view.DrawingPanel;

public class MouseController  extends MouseAdapter {
	Model model;
	DrawingPanel panel;
	public MouseController(Model m, DrawingPanel p) {
		this.model = m;
		this.panel = p;
	}
	
	@Override
    public void mousePressed(MouseEvent e) {
		if (true == model.getHasWon()) {
			return;
		}
		Point p = e.getPoint();
		model.selectMouseNode(p.x, p.y);
		panel.repaint();
	}
}
