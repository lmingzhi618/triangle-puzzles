package controller;

import model.Model;
import view.TrianglePuzzleApp;

public class ResetController {
	Model model;
	TrianglePuzzleApp app;
	
	public ResetController(TrianglePuzzleApp app, Model m) {
		this.model = m;
		this.app = app;
	}
	
	public void process() {
		System.out.println("unselect all clicked");
		model.reset();
		app.getCongratsLabel().setText("");
		app.getSwapBtn().setEnabled(true);
		app.getUnselectAllBtn().setEnabled(true);
		int nMove = model.getNumberMove();
		int nScore = model.getScore();
		app.getMoveField().setText(String.valueOf(nMove));;
		app.getScoreField().setText(String.valueOf(nScore));;
		app.getPanel().repaint();
	}
}
