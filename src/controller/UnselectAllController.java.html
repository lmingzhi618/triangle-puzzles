package controller;

import model.Model;
import view.TrianglePuzzleApp;

public class UnselectAllController {
	Model model;
	TrianglePuzzleApp app;
	
	public UnselectAllController(TrianglePuzzleApp app, Model m) {
		this.model = m;
		this.app = app;
	}
	
	public void process() {
		model.unselectAllNodes();
		app.getPanel().repaint();
	}
}
