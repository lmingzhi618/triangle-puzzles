package model;

public class Node implements Cloneable, Comparable<Node>
{
	int x;
	int y;
	int width = 20;
	int idx;			// node�±�
	boolean isSelected = false;
	
	public Node(int x, int y, int idx) {
		this.x = x;
		this.y = y;
		this.idx = idx;
	}
	
	public boolean isTargeted(int x, int y) {
		if ((x >= (this.x-width) && x <= (this.x + width)) && 
			(y >= (this.y-width) && y <= (this.y + width))) {
			return true;
		}
		return false;
	}
	
	public void toogleSelected() {
		if (true == this.isSelected) {
			this.isSelected = false;
			return;
		}
		this.isSelected = true;
	}
	
    public Object clone() {    
        Object o = null;    
        try {    
            o=(Node)super.clone();
        } catch(CloneNotSupportedException e) {    
             System.out.println(e.toString());    
        }    
        return o;    
    }
    
    public int getX() {
    	return this.x;
    }

    public int getY() {
    	return this.y;
    }
    
    public int getWidth() {
    	return this.width;
    }
    
    public boolean isNodeSelected() {
    	return isSelected;
    }
    
    public int compareTo(Node n) {
    	return this.idx - n.idx;
    }
    
    public int getIdx() {
    	return this.idx;
    }
}
