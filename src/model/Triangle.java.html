package model;

import java.awt.Color;

public class Triangle {
	Edge e1;
	Edge e2;
	Edge e3;
	Color winColor;
	
	Triangle(Edge e1, Edge e2, Edge e3, Color c) {
		this.e1 = e1;
		this.e2 = e2;
		this.e3 = e3;
		this.winColor = c;
	}
	
	public boolean isAllEdgesSameColor() {
		if (e1.color == e2.color &&
			e1.color == e3.color) {
			return true;
		};
		return false;
	}
	
	boolean hasEdge(Edge e) {
		if (this.e1.idx == e.idx || 
			this.e2.idx == e.idx || 
			this.e3.idx == e.idx) {
			return true;
		}
		return false;
	}
	
	boolean hasWinState() {
		if (false == isAllEdgesSameColor()) {
			return false;
		}
		if (this.winColor != this.e1.color) {
			return false;
		}
		return true;
	}
}
