package model;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

public class NodeTest {
	private int x = 165;
  	private int y = 13;
  	private int width = 20;
	
  	@Test
	void testIsTargeted() {
	  	Node node = new Node(x, y, 0);
		assertFalse(node.isTargeted(0, 0));
		assertFalse(node.isTargeted(x-25, y-25));
		assertTrue(node.isTargeted(x-5, y-5));
	}

	@Test
	void testToogleSelected() {
		Node node = new Node(x, y, 0);
		assertFalse(node.isNodeSelected());
		node.toogleSelected();
		assertTrue(node.isNodeSelected());
	}

	@Test
	void testClone() {
		Node node = new Node(x, y, 0);
		assertNotNull(node.clone());
	}

	@Test
	void testGetX() {
		Node node = new Node(x, y, 0);
		assertEquals(x, node.getX());
	}
	
	@Test
	void testGetY() {
	  	Node node = new Node(x, y, 0);
		assertEquals(y, node.getY());
	}
	
	@Test
	void testGetWidth() {
	  	Node node = new Node(x, y, 0);
		assertEquals(width, node.getWidth());
	}
	
	@Test
	void testIsNodeSelected() {
	  	Node node = new Node(x, y, 0);
		assertFalse(node.isNodeSelected());
		node.toogleSelected();
		assertTrue(node.isNodeSelected());
	}
	
	@Test
	void testCompareTo() {
	  	Node node = new Node(x, y, 0);
		assertEquals(0, node.compareTo(node));
	}

	@Test
	void testGetIdx() {
		Node node = new Node(x, y, 0);
		assertEquals(0, node.getIdx());
	}
}
