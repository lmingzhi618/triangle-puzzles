package model;

import static org.junit.jupiter.api.Assertions.*;

import java.awt.Color;

import org.junit.jupiter.api.Test;

public class TriangleTest {
	private	int d = 50;
	private	int x = 165;
	private	int y = 13;
	private Node n0 = new Node(x, y, 0);
	private Node n1 = new Node(x-d, y+d*2, 1);
	private Node n2 = new Node(x+d, y+d*2, 2);
    private Node n3 = new Node(x-d*2, y+d*4, 3);

    private Edge e01 =  new Edge(n0, n1, Color.RED, 1);
    private Edge e02 =  new Edge(n0, n2, Color.RED, 2);
    private Edge e12 =  new Edge(n1, n2, Color.GREEN, 12);
    
    private Edge e13 =  new Edge(n1, n3, Color.RED, 13);

    @Test 
    void testIsAllEdgesSameColoretIdx() {
		Triangle t = new Triangle(e01, e02, e12, Color.RED);
        assertFalse(t.isAllEdgesSameColor());
    }

    @Test 
    void testhasEdge() {
		Triangle t = new Triangle(e01, e02, e12, Color.RED);
        assertTrue(t.hasEdge(e01));
        assertFalse(t.hasEdge(e13));
    }

    @Test
    void testHasWinState() {
		Triangle t = new Triangle(e01, e02, e12, Color.RED);
        assertFalse(t.hasWinState());
        e12.setColor(Color.RED);
        assertTrue(t.hasWinState());
    }
}
