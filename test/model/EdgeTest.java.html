package model;

import static org.junit.jupiter.api.Assertions.*;

import java.awt.Color;

import org.junit.jupiter.api.Test;

public class EdgeTest {
	private	int d = 50;
	private	int x = 165;
	private	int y = 13;
    private int idx = 1;
	private Node n1 = new Node(x, y, 0);
	private Node n2 = new Node(x-d, y+d*2, 1);
    
    @Test 
    void testGetIdx() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        assertEquals(idx, e.getIdx());
    }
    
    @Test 
    void testSetColor() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        e.setColor(Color.GREEN);
        assertEquals(Color.GREEN, e.getColor());
    }
    
    @Test 
    void testGetColor() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        assertEquals(Color.RED, e.getColor());
    }
    
    @Test 
    void testClone() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        assertNotNull(e.clone());
    }
    
    @Test 
    void testGetNodes() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        assertEquals(2, e.getNodes().length);
    }
    @Test 
    void testCompareTo() {
        Edge e = new Edge(n1, n2, Color.RED, 1);
        assertEquals(0, e.compareTo(e));
    }
}
