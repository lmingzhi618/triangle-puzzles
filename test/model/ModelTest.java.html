		Node n2 = new Node(x+d, y+d*2, 2);
        Edge e1 = new Edge(n0, n1, Color.RED, 1);
        Edge e2 = new Edge(n0, n2, Color.RED, 2);

		Node n3 = new Node(x-d*2, y+d*4, 3);
        Edge e1 = new Edge(n0, n1, Color.RED, 1);
        Edge e2 = new Edge(n0, n2, Color.RED, 2);
        Edge e12 = new Edge(n1, n2, Color.GREEN, 12);
        Edge e13 = new Edge(n1, n3, Color.RED, 13);
    
        Model m = new Model();
