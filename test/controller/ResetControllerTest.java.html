package controller;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

import model.Model;
import view.TrianglePuzzleApp;

public class ResetControllerTest {
	private int d = 50;
	private int x = 165;
	private int y = 13;
	
	@Test
    void testProcess() {
		TrianglePuzzleApp app = new TrianglePuzzleApp();
        Model m = new Model();
        
        m.selectMouseNode(x, y);
        m.selectMouseNode(x-d, y+d*2);
        m.selectMouseNode(x+d, y+d*2);
        assertEquals(3, m.isSelectedNodesAvailable().size());
        ResetController c = new ResetController(app, m);
        c.process();
    }
}
