package controller;

import static org.junit.jupiter.api.Assertions.*;

import java.awt.event.InputEvent;
import java.awt.event.MouseEvent;

import org.junit.jupiter.api.Test;

import model.Model;
import view.DrawingPanel;
import view.TrianglePuzzleApp;

public class MouseControllerTest {
	private int x = 165;
	private int y = 13;
    
	@Test 
    void testMousePressed() {
		Model m = new Model();
		TrianglePuzzleApp app = new TrianglePuzzleApp();
        DrawingPanel panel = app.getPanel();
		MouseController c = new MouseController(m, panel);
		panel.addMouseListener(new MouseController(m, panel));
		assertEquals(0, m.isSelectedNodesAvailable().size());
		MouseEvent e = new MouseEvent(
		            panel,                                      
		            MouseEvent.MOUSE_CLICKED,
		            System.currentTimeMillis(),
		            InputEvent.BUTTON1_MASK,
		            x, y,
		            1, // click count
		            false
		        );
		c.mousePressed(e);
	}
}
